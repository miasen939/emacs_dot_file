  (use-package org-download
    :after org
    :bind (:map org-mode-map
                ("C-c C-M-y" . org-download-clipboard)   ; 推荐快捷键：粘贴剪贴板图片

    ;; 核心配置写在这里

    ;; Mac/Linux 剪贴板截图特别推荐再加这一行
